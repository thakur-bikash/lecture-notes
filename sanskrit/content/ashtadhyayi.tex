\setchapterhead{अष्टाध्यायी - संज्ञाप्रकरणम्}


\begin{sutra}{१.१.१}
वृद्धिरादैच्
\end{sutra}

\begin{anuvritti}
-
\end{anuvritti}
%\begin{adhikar}
%-
% \end{adhikar}
\begin{padartha}
\textbf{अन्वयः} वृद्धिः आत्  ऐच् \\[4pt]
\textbf{पदपरिचयः:}
\begin{itemize}
    \item \textbf{वृद्धिः} —  (प्रथमा-एकवचनम्) 
    \item \textbf{आत्-ऐच्} — (प्रथमा-एकवचनम्)

\end{itemize}
\end{padartha}

\begin{vyakhya}[व्याख्या]
 आ, ऐ, औ - एतेषां त्रयाणां 'वृद्धिः' इति संज्ञा भवति ।  
  आदैच्च वृद्धिसंज्ञः स्यात् ॥
  
\end{vyakhya}
