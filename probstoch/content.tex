\section{PART I --- FOUNDATIONS}

\subsection{1. Probability Spaces and Random Variables}
\begin{itemize}
\item Probability spaces; $\sigma$-algebras; Kolmogorov axioms
\item Random variables; distribution functions; pushforward measures
\item Expectation via Lebesgue integration; moment generating functions
\item Independence; product spaces; Kolmogorov extension theorem
\end{itemize}

\subsection{2. Classical Limit Theorems}
\begin{itemize}
\item Laws of large numbers: weak (Chebyshev) and strong (Borel-Cantelli)
\item Characteristic functions; L\'evy continuity theorem
\item Central limit theorem: Lindeberg-Feller condition; Berry-Esseen rates
\item Infinitely divisible distributions; stable laws (brief)
\end{itemize}

\subsection{3. Conditional Expectation and Martingales}
\begin{itemize}
\item Conditional expectation as $L^2$ projection; properties; tower property
\item Discrete-time martingales; stopping times; filtrations
\item Optional stopping theorem; Doob's inequalities (maximal, $L^2$, upcrossing)
\item Convergence theorems: martingale convergence; uniform integrability
\end{itemize}

\subsection{4. Markov Chains (Discrete Time)}
\begin{itemize}
\item Transition kernels; Chapman-Kolmogorov equations
\item Recurrence and transience; hitting times; classification of states
\item Stationary distributions; detailed balance; ergodic theorem
\item Mixing times; spectral gap; connection to spectral graph theory (Course 2)
\end{itemize}

\section{PART II --- CONTINUOUS-TIME PROCESSES}

\subsection{5. Brownian Motion}
\begin{itemize}
\item Construction: L\'evy-Ciesielski and Kolmogorov approaches
\item Path properties: continuity, nowhere differentiability, quadratic variation
\item Strong Markov property; reflection principle; hitting times
\item Connections to harmonic functions; Kakutani's theorem
\end{itemize}

\subsection{6. Stochastic Calculus}
\begin{itemize}
\item It\^o integral: $L^2$ construction; isometry property
\item It\^o's formula: single and multi-dimensional
\item SDEs: existence and uniqueness (Lipschitz coefficients)
\item Girsanov theorem; Novikov condition; Cameron-Martin
\end{itemize}

\subsection{7. Diffusion Processes and Generators}
\begin{itemize}
\item Generator of a diffusion; Kolmogorov forward (Fokker-Planck) and backward equations
\item Stationary distributions; conditions for existence and uniqueness
\item Connections to semigroup theory; Feynman-Kac formula
\item Diffusions on manifolds (brief)
\end{itemize}

\subsection{8. L\'evy Processes and Jump Processes}
\begin{itemize}
\item Characteristic exponent; L\'evy-Khintchine representation
\item Poisson random measures; compensation
\item Stochastic integrals with jumps; It\^o formula with jumps
\item Applications: finance; insurance ruin; queueing
\end{itemize}

\subsection{9. Gaussian Processes}
\begin{itemize}
\item Covariance functions; mean functions; RKHS connection
\item Sample path regularity via Kolmogorov-Chentsov
\item Gaussian process regression; posterior computation; marginal likelihood
\item Applications: Bayesian optimization; probabilistic numerics (Course 4)
\end{itemize}

\section{PART III --- ERGODIC THEORY AND MIXING}

\subsection{10. Ergodic Theory}
\begin{itemize}
\item Measure-preserving systems; ergodicity; invariant measures
\item Von Neumann $L^2$ ergodic theorem
\item Birkhoff pointwise ergodic theorem
\item Mixing conditions; mixing for Markov chains and diffusions
\item Applications: time series; statistical mechanics; MCMC convergence
\end{itemize}

\subsection{11. Convergence Rates and Mixing}
\begin{itemize}
\item Total variation distance; coupling inequality
\item Poincar\'e inequality; spectral gap; Cheeger inequality connection
\item Log-Sobolev inequalities; hypercontractivity; Bakry-\'Emery criterion
\item Coupling methods; maximal coupling; Lindvall's coupling
\end{itemize}

\section{PART IV --- COMPUTATIONAL PROBABILITY}

\subsection{12. Monte Carlo Methods}
\begin{itemize}
\item Classical Monte Carlo; variance reduction (control variates, stratification, antithetics)
\item Metropolis-Hastings; Gibbs sampler; detailed balance
\item Hamiltonian Monte Carlo; MALA; NUTS
\item Convergence diagnostics; effective sample size; Gelman-Rubin
\end{itemize}

\subsection{13. Sequential Monte Carlo and Filtering}
\begin{itemize}
\item Particle filters; sequential importance resampling; degeneracy
\item Kalman filter; Kalman-Bucy filter; derivation and optimality
\item Nonlinear filtering; Zakai equation; Kushner-Stratonovich
\item Applications: robotics (SLAM), weather forecasting, finance
\end{itemize}

\subsection{14. Large Deviations for Processes}
\begin{itemize}
\item Cram\'er's theorem on path spaces; Schilder's theorem for BM
\item Freidlin-Wentzell theory: quasi-potential; rare events for diffusions
\item Applications: MCMC efficiency; transition state theory; rare event simulation
\end{itemize}

\subsection{15. Stochastic Optimization and Sampling}
\begin{itemize}
\item Langevin dynamics; overdamped vs.\ underdamped; connections to MCMC
\item SGD as stochastic differential equation; continuous-time limit
\item Stein variational gradient descent
\item Interacting particle systems; propagation of chaos; McKean-Vlasov limits
\end{itemize}
