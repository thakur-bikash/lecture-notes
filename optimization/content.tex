\section{PART I --- CONVEX ANALYSIS AND DUALITY}

\subsection{1. Convex Sets and Functions}
\begin{itemize}
\item Convex sets: examples, operations, separation theorems
\item Convex functions: characterizations (epigraph, first-order, second-order)
\item Examples: norms, log-sum-exp, entropy, matrix functions (log det, trace exp)
\item Generalized inequalities; cones; proper cones
\end{itemize}

\subsection{2. Duality Theory}
\begin{itemize}
\item Lagrangian; Lagrange dual function; weak duality
\item Slater's condition; strong duality
\item KKT conditions: necessary and (under convexity) sufficient
\item Fenchel duality; primal-dual relationships
\end{itemize}

\subsection{3. Subdifferentials and Non-Smooth Analysis}
\begin{itemize}
\item Subdifferential of a convex function; examples ($|x|$, $\ell_1$ norm)
\item Calculus rules: sum, chain rule, conjugate; Moreau-Rockafellar theorem
\item Optimality conditions with subdifferentials
\item Proximal operator: definition, properties, closed-form examples
\end{itemize}

\subsection{4. Linear Programming}
\begin{itemize}
\item Standard form; geometry of polytopes; vertices and basic feasible solutions
\item Simplex method: algebraic view; pivot rules; complexity
\item Duality in LP; complementary slackness; dual simplex
\item Network flow problems; transportation; assignment
\end{itemize}

\subsection{5. Conic Programming}
\begin{itemize}
\item Second-order cone programming (SOCP): examples; duality; applications
\item Semidefinite programming (SDP): PSD cone; dual; KKT conditions
\item Interior point methods for conic programs
\item Applications: eigenvalue optimization, sum-of-squares relaxations, robust optimization
\end{itemize}

\section{PART II --- GRADIENT-BASED ALGORITHMS}

\subsection{6. Gradient Descent}
\begin{itemize}
\item Gradient descent: Lipschitz gradient; $O(1/k)$ convergence for convex objectives
\item Linear convergence under strong convexity; rate constants
\item Subgradient methods for non-smooth problems; $O(1/\sqrt{k})$ convergence
\item Projected gradient descent; projection onto convex sets
\end{itemize}

\subsection{7. Accelerated and Proximal Methods}
\begin{itemize}
\item Nesterov acceleration: $O(1/k^2)$ optimal rate; momentum interpretation
\item Proximal gradient method (ISTA); convergence analysis
\item FISTA: accelerated proximal gradient
\item Composite objectives $f(x)+g(x)$; splitting structure
\end{itemize}

\subsection{8. ADMM and Operator Splitting}
\begin{itemize}
\item Augmented Lagrangian; ADMM derivation
\item Convergence analysis; sublinear and linear rates
\item Distributed ADMM; consensus formulation
\item Applications: LASSO, image processing, graph problems
\end{itemize}

\subsection{9. Newton and Quasi-Newton Methods}
\begin{itemize}
\item Newton's method: local quadratic convergence; Newton decrement
\item Quasi-Newton: secant condition; BFGS update; convergence theory
\item L-BFGS: limited memory; implementation; convergence
\item Gauss-Newton; Levenberg-Marquardt for nonlinear least squares
\end{itemize}

\subsection{10. Interior Point Methods}
\begin{itemize}
\item Logarithmic barrier; central path; path-following
\item Primal-dual interior point: predictor-corrector algorithm
\item Polynomial complexity analysis
\item Self-concordant functions (Nesterov-Nemirovskii theory, brief)
\end{itemize}
